三、启动命令
------------------------------------------------------------

# 1. 构建镜像
cd ~/.openclaw/workspace/agents/workspace-xuanyuan/boss-recruitment-system/
docker-compose build

# 2. 启动服务
docker-compose up -d

# 3. 查看状态
docker-compose ps
docker-compose logs -f

# 4. 停止服务
docker-compose down
